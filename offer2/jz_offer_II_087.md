var restoreIpAddresses = function (s) {
	let res = [];
	let track = [];

	const isValid = (str) => {
		if ((str.length > 1 && str[0] == '0') || Number(str) > 255) {
			return false;
		}
		return true;
	};

		if (track.length > 4) return;

			res.push(track.join('.'));
			return;
		}

			const str = s.substring(start, i + 1);
			if (isValid(str)) {

				backtrack(i + 1);

			}
		}
	};

	return res;
};
