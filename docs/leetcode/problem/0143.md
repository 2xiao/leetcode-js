 * Definition for singly-linked list.
 * function ListNode(val, next) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.next = (next===undefined ? null : next)
 * }

/**
 * @param {ListNode} head
 * @return {void} Do not return anything, modify head in-place instead.
 */
var reorderList = function (head) {
  if (!head || !head.next) return head;

  // ÂØªÊâæ‰∏≠Èó¥ËäÇÁÇπ
  let slow = head;
  let fast = head;
  while (fast && fast.next && fast.next.next) {
    slow = slow.next;
    fast = fast?.next?.next;

  // ÂèçËΩ¨Âêé‰∏ÄÂçäÈìæË°®Ôºåeg: 1->2->3->4->5->6 to 1->2->3->6->5->4

  let cur = middle.next;
  while (cur.next) {
    let temp = cur.next;
    cur.next = temp.next;
    temp.next = middle.next;
    middle.next = temp;
  }

  // ÊåâË¶ÅÊ±ÇÈáçÊñ∞ÊãºÊé•ÈìæË°®Ôºåeg: 1->2->3->6->5->4 to 1->6->2->5->3->4

  let p4 = middle.next;
  while (p3 != middle) {
    middle.next = p4.next;
    p4.next = p3.next;
    p3.next = p4;
    p3 = p4.next;
    p4 = middle.next;
  }
};
