/**
 * @param {string} s
 * @return {number}
 */
var romanToInt = function (s) {
  const roman = {
    I: 1,
    V: 5,
    X: 10,
    L: 50,
    C: 100,
    D: 500,
    M: 1000,
  };

  let num, nextNum;
  let total = 0;

  for (let i = 0; i < s.length; i++) {
    num = roman[s[i]];
    nextNum = roman[s[i + 1]];
    if (num < nextNum) {
      total -= num;
    } else {
      total += num;
    }
  }

  return total;
};
