/**
 * @param {string} s
 * @return {number}
 */
var myAtoi = function (s) {
  s = s.trim();
  if (s.length == 0) return 0;

  let i = 0;
  let res = 0;
  let sign = 1;

  if (s[i] === "-") {
    i++;
    sign = -1;
  } else if (s[i] === "+") {
    i++;
  }

  while ((i < s.length) && !isNaN(parseInt(s[i]))) {
    res = 10 * res + parseInt(s[i]);
    i++;
  }

  res *= sign;
  res = Math.max(Math.min(2 ** 31 - 1, res), -(2 ** 31));
  return res;
};
