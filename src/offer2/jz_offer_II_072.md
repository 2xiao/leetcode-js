var mySqrt = function (x) {
	if (x == 1 || x == 0) return x;
	let left = 0,
		right = x,
		res;
	while (left <= right) {
		let mid = ((left + right) / 2) | 0;
		if (mid * mid > x) {
			right = mid - 1;
		} else {
			res = mid;
			left = mid + 1;
		}
	}
	return res;
};
