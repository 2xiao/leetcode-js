/**
 * @param {number} num
 * @return {string}
 */
var intToRoman = function (num) {
  const int = [1000, 900, 500, 400, 100, 90, 50, 40, 10, 9, 5, 4, 1];
  const roman = [
    "M",
    "CM",
    "D",
    "CD",
    "C",
    "XC",
    "L",
    "XL",
    "X",
    "IX",
    "V",
    "IV",
    "I",
  ];
  let res = "";
  for (let i = 0; i < int.length; i++) {
    while (num >= int[i]) {
      res += roman[i];
      num -= int[i];
    }
  }
  return res;
};
