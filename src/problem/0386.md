/**
 * @param {number} n
 * @return {number[]}

var lexicalOrder = function (n) {
	let res = [];
	let num = 1;
	while (res.length < n) {
		res.push(num);
		if (num * 10 <= n) {
			num *= 10;
		} else {
			while (num % 10 == 9 || num >= n) {
				num = Math.floor(num / 10);
			}
			num++;
		}
	}
	return res;
};
