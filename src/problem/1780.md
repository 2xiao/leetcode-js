> Input: n = 12

> Output: true

> Input: n = 91
>
> Output: true
>
> Explanation: 91 = 30 + 32 + 34

>
> Output: false

**Constraints:**

/**

 * @return {boolean}

var checkPowersOfThree = function (n) {
	while (n > 0) {
		if (n % 3 == 2) return false;
		n = (n / 3) | 0;
	}
	return true;
};
