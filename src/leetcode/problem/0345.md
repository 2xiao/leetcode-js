/**
 * @param {string} s
 * @return {string}
 */
var reverseVowels = function (s) {
  s = s.split("");
  let left = 0;
  let right = s.length - 1;
  while (left < right) {
    if (!isVowel(s[left])) {
      left++;
    } else if (!isVowel(s[right])) {
      right--;
    } else {
      let temp = s[left];
      s[left] = s[right];
      s[right] = temp;
      left++;
      right--;
    }
  }
  return s.join("");
};

var isVowel = function (i) {
  i = i.toLowerCase();
  return i == "a" || i == "e" || i == "i" || i == "o" || i == "u";
};
