/**
 * @param {ListNode} head
 * @param {number} val
 * @return {ListNode}
 */
var deleteNode = function (head, val) {
	if (head.val == val) return head.next;
	let prev = head;
	while (prev.next) {
		if (prev.next.val == val) {
			prev.next = prev.next.next;
			return head;
		}
		prev = prev.next;
	}
};
