var convert = function (s, numRows) {
  let arr = new Array(numRows).fill("");
  // è´Ÿè´£ä»Žç¬¬ä¸€è¡Œå¼€å§‹å‘ä¸‹åŠ ï¼Œç›´åˆ°ç¬¬numRowsè¡Œ
  let down = 0;

  let up = numRows - 2;
  for (let i = 0; i < s.length; i++) {
    if (down < numRows) {
      arr[down] += s[i];
      down++;
    } else if (up > 0) {
      arr[up] += s[i];
      up--;
    } else {
      arr[0] += s[i];
      down = 1;
      up = numRows - 2;
    }
  }
  return arr.join("");
};
