```javascript
/**
 * @param {number} x
 * @return {number}
 */
var reverse = function (x) {
  const isNegative = x > 0 ? 1 : -1;
  let temp = 0;

  x *= isNegative;

  while (x > 0) {
    temp = temp * 10 + (x % 10);
    x = Math.floor(x / 10);

    if (temp < -(2 ** 31) || temp > 2 ** 31 - 1) return 0;
  }

  return temp * isNegative;
};
```
