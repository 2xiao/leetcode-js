```javascript
/**
 * @param {number[]} nums
 * @return {number}
 */
var majorityElement = function (nums) {
  let target;
  let count = 0;
  for (let item of nums) {
    if (count === 0 || item === target) {
      count++;
      target = item;
    } else {
      count--;
    }
  }
  return target;
};
```
