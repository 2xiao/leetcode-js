var generateMatrix = function (n) {
	let count = 0;
	let left = 0;
	let right = n - 1;
	let top = 0;
	let bottom = n - 1;
	let res = new Array(n).fill(0).map(() => new Array(n));

	while (count < n * n) {
		// åœ¨ä¸Šè¾¹ç•Œå‘å³æ‰«æ
