var takeAttendance = function (records) {
	let left = 0;
	let right = records.length - 1;
	while (left <= right) {
		const mid = Math.floor((right + left) / 2);
		if (mid == records[mid]) {
			left = mid + 1;
		} else {
			right = mid - 1;
		}
	}
	return left;
};
