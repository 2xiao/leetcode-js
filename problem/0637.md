```javascript
/**
 * @param {TreeNode} root
 * @return {number[]}
 */
var averageOfLevels = function (root) {
	let res = [];
	if (!root) return res;
	let queue = [root];
	while (queue.length) {
		let len = queue.length;
		let temp = 0;
		for (let i = 0; i < len; i++) {
			if (queue[i].left) queue.push(queue[i].left);
			if (queue[i].right) queue.push(queue[i].right);
			temp += queue[i].val;
		}
		queue = queue.slice(len);
		res.push(temp / len);
	}
	return res;
};
```
