	const getReqNum = (first, n) => {
		let gap = 0,
			last = first + 1;
		while (first <= n) {
			gap += Math.min(n + 1, last) - first;

			last *= 10;
		}

		let req = getReqNum(num, n);

			i += req;
			num++;
		} else {
			i++;
			num *= 10;
		}
	}

	return num;
};
