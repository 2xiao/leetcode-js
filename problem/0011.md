var maxArea = function (height) {
	let left = 0;
	let right = height.length - 1;
	let _height, width;
	let max = 0;
	while (left < right) {
		width = right - left;
		if (height[left] > height[right]) {
			_height = height[right];
			right--;
		} else {
			_height = height[left];
			left++;
		}
		max = Math.max(max, _height * width);
	}
	return max;
};
