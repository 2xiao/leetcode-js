/**
 * @param {number[]} score
 * @return {string[]}
 */
var findRelativeRanks = function (score) {
	const rank = [...score].sort((a, b) => b - a);

	let map = new Map();
	rank.map((val, index) => {
		if (index == 0) {
			map.set(val, 'Gold Medal');
		} else if (index == 1) {
			map.set(val, 'Silver Medal');
		} else if (index == 2) {
			map.set(val, 'Bronze Medal');
		} else {
			map.set(val, String(index + 1));
		}
	});

	return score.map((i) => map.get(i));
};
