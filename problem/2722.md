var join = function (arr1, arr2) {
	let dict = {};
	arr2.forEach((item) => (dict[item.id] = item));

	arr1.forEach((item) => {
		if (dict[item.id]) {
			dict[item.id] = { ...item, ...dict[item.id] };
		} else {
			dict[item.id] = item;
		}
	});
	return Object.values(dict);
};
