/**
 * @param {number[]} nums
 * @return {number}
 */
var majorityElement = function (nums) {
	let major;
	let count = 0;
	for (let item of nums) {
		if (count === 0 || item === major) {
			count++;
			major = item;
		} else {
			count--;
		}
	}
	return major;
};
