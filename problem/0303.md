/**
 * @param {number[]} nums
 */
class NumArray {
	constructor(nums) {
		this.preSum = [0];
		for (let i = 1; i <= nums.length; i++) {
			this.preSum[i] = this.preSum[i - 1] + nums[i - 1];
		}
	}

	/**
	 * @param {number} left
	 * @param {number} right
	 * @return {number}
	 */
	sumRange(left, right) {
		return this.preSum[right + 1] - this.preSum[left];
	}
}
