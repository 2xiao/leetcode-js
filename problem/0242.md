var isAnagram = function (s, t) {
	let map = new Map();
	for (let i of s) {
		let val = map.get(i) || 0;
		map.set(i, val + 1);
	}
	for (let j of t) {
		let val = map.get(j) || 0;
		map.set(j, val - 1);
	}
	return [...map.values()].filter((i) => i != 0).length === 0;
};
