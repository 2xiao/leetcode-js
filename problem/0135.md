/**
 * @param {number[]} ratings
 * @return {number}
 */
var candy = function (ratings) {
	const n = ratings.length;
	let candy = new Array(n).fill(1);

	for (let i = 1; i < n; i++) {
		if (ratings[i] > ratings[i - 1] && candy[i] <= candy[i - 1]) {
			candy[i] = candy[i - 1] + 1;
		}
	}

	for (let i = n - 2; i >= 0; i--) {
		if (ratings[i] > ratings[i + 1] && candy[i] <= candy[i + 1]) {
			candy[i] = candy[i + 1] + 1;
		}
	}

	return candy.reduce((acc, cur) => acc + cur, 0);
};
