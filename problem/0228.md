/**
 * @param {number[]} nums
 * @return {string[]}
 */
var summaryRanges = function (nums) {
	let set = new Set(nums),
		res = [];
	for (let i = 0; i < nums.length; i++) {
		let num = nums[i],
			str = '' + num;
		if (!set.has(num + 1)) {
			res.push(str);
		} else {
			let len = 1;
			while (set.has(num + len)) {
				len++;
				i++;
			}
			res.push(str + '->' + (num + len - 1));
		}
	}
	return res;
};
