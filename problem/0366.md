/**
 * @param {TreeNode} root
 * @return {number[][]}
 */
var findLeaves = function (root) {
	if (!root) return [];
	let res = [];
	const maxDepth = (root) => {
		if (!root) return 0;
		let depth = Math.max(maxDepth(root.left), maxDepth(root.right)) + 1;
		res[depth - 1].push(root.val);
		return depth;
	};
	maxDepth(root);
	return res;
};
