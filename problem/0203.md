```javascript
/**
 * @param {ListNode} head
 * @param {number} val
 * @return {ListNode}
 */
var removeElements = function (head, val) {
	if (!head) return head;
	let res = new ListNode(0, head);
	let prev = res;
	while (prev.next) {
		if (prev.next.val === val) {
			prev.next = prev.next.next;
		} else {
			prev = prev.next;
		}
	}
	return res.next;
};
```
