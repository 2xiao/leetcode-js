var postorder = function (root) {
	if (!root) return [];
	let res = [];
	for (let i of root.children) {
		res.push(...postorder(i));
	}
	res.push(root.val);
	return res;
};

/**
 * @param {Node|null} root

var postorder = function (root) {
	if (!root) return [];
	let stack = [root];
	let res = [];
	while (stack.length) {
		let node = stack.pop();
		res.unshift(node.val);
		for (let i of node.children) {
			stack.push(i);
		}
	}
	return res;
};
