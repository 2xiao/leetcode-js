/**

var sumNumbers = function (root) {
	const dfs = (root, currentSum) => {
		if (!root) return 0;
		currentSum = currentSum * 10 + root.val;
		if (!root.left && !root.right) return currentSum;
		let left = dfs(root.left, currentSum);
		let right = dfs(root.right, currentSum);
		return left + right;
	};
	return dfs(root, 0);
};
