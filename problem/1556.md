```javascript
/**
 * @param {number} n
 * @return {string}
 */
var thousandSeparator = function (n) {
	n = '' + n;
	let p = n.length - 1;
	while (p > 2) {
		n = n.slice(0, p - 2) + '.' + n.slice(p - 2);
		p -= 3;
	}
	return n;
};
```
