var originalDigits = function (s) {
	let count = {};
	for (let char of s) {
		if (!(char in count)) {
			count[char] = 1;
		} else {
			count[char]++;
		}
	}
	let res = [];
	while (count['z']) {
		res.push(0);
		count['z']--;
		count['e']--;
		count['r']--;
		count['o']--;
	}
	while (count['w']) {
		res.push(2);
		count['t']--;
		count['w']--;
		count['o']--;
	}
	while (count['x']) {
		res.push(6);
		count['s']--;
		count['i']--;

	}
	while (count['g']) {
		res.push(8);
		count['e']--;
		count['i']--;
		count['g']--;

		count['t']--;
	}
	while (count['t']) {
		res.push(3);
		count['t']--;
		count['h']--;

		count['e'] -= 2;
	}
	while (count['r']) {
		res.push(4);

	}
	while (count['f']) {
		res.push(5);

	while (count['o']) {
		res.push(1);
		count['o']--;
		count['n']--;
		count['e']--;
	}
	return res.sort().join('');
