/**
 * @param {TreeNode} root
 * @return {string[]}
 */
var binaryTreePaths = function (root) {
	let res = [];
	if (!root) return res;
	if (!root.left && !root.right) return ['' + root.val];
	let tempLeft = binaryTreePaths(root.left);
	for (let i of tempLeft) {
		res.push(root.val + '->' + i);
	}
	let tempRight = binaryTreePaths(root.right);
	for (let i of tempRight) {
		res.push(root.val + '->' + i);
	}
	return res;
};
