 * @param {number[]} citations
 * @return {number}

var hIndex = function (citations) {
	let n = citations.length,
		left = 0,
		right = n - 1;
	while (left <= right) {
		const mid = (left + right) >> 1;
		if (citations[mid] >= n - mid) {
			right = mid - 1;
		} else {
			left = mid + 1;
		}
	}
	return n - left;
};
