var isPalindrome = function (head) {
	if (!head || !head.next) return true;

	// æ‰¾åˆ°ä¸­ç‚¹
	let slow = head;
	let fast = head;
	while (fast.next && fast.next.next) {

		fast = fast.next.next;
	}

	while (cur.next) {
		let temp = cur.next;
		cur.next = temp.next;
		temp.next = middle.next;
		middle.next = temp;
	}

	slow = head;
	fast = middle.next;
	while (fast) {
		if (slow.val == fast.val) {
			slow = slow.next;
			fast = fast.next;
		} else {
			return false;
		}
	}
	return true;
};
