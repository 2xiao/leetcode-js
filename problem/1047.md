```javascript
/**
 * @param {string} s
 * @return {string}
 */
var removeDuplicates = function (s) {
	let stack = [];
	for (let i = 0; i < s.length; i++) {
		let len = stack.length;
		if (len != 0 && stack[len - 1] == s[i]) {
			stack.pop();
		} else {
			stack.push(s[i]);
		}
	}
	return stack.join('');
};
```
