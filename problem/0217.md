```javascript
/**
 * @param {number[]} nums
 * @return {boolean}
 */
var containsDuplicate = function (nums) {
	const map = new Map();
	for (let item of nums) {
		if (map.has(item)) return true;
		map.set(item, 1);
	}
	return false;
};
```
