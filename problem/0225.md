class MyStack {
	constructor() {
		this.stack = [];
	}
	// @param {number} x
	// @return {void}
	push(x) {
		this.stack.push(x);
		for (let i = 0; i < this.stack.length - 1; i++) {
			this.stack.push(this.stack.shift());
		}
	}

	// @return {number}
	pop() {
		return this.stack.shift();
	}

	// @return {number}
	top() {
		return this.stack[0];
	}

	// @return {boolean}
	empty() {
		return this.stack.length === 0;
	}
}
