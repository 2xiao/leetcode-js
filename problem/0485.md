> Input: nums = [1,1,0,1,1,1]

**Constraints:**

- `1 <= nums.length <= 10^5`

```javascript
/**
 * @param {number[]} nums
 * @return {number}
 */
var findMaxConsecutiveOnes = function (nums) {
	let max = 0;
	let count = 0;
	for (let i = 0; i < nums.length; i++) {
		if (nums[i] == 1) {
			count++;
		} else {
			count = 0;
		}
		max = Math.max(max, count);
	}
	return max;
};
```
