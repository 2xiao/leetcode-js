var change = function (amount, coins) {

		for (let j = 1; j <= amount; j++) {

				dp[i][j] = dp[i - 1][j];
			}
		}
	}

};

	const dp = new Array(amount + 1).fill(0);
	dp[0] = 1;
	for (let coin of coins) {
		for (let j = 1; j <= amount; j++) {

		}
	}
