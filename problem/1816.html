 * @param {string} s
 * @param {number} k
 * @return {string}
 */
var truncateSentence = function (s, k) {
	let wordCount = 0;
	let result = '';

	for (let i = 0; i < s.length; i++) {
		if (s[i] === ' ' && wordCount === k) break;
		if (s[i] === ' ') wordCount++;
		if (wordCount < k) result += s[i];
	}

	return result;
};
