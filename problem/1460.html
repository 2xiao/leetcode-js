var canBeEqual = function (target, arr) {
	let count = new Map();
	for (let num of target) {
		count.set(num, (count.get(num) || 0) + 1);
	}

	for (let num of arr) {
		count.set(num, (count.get(num) || 0) - 1);
	}

	for (let value of count.values()) {

	}
	return true;
