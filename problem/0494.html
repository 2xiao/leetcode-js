var findTargetSumWays = function (nums, target) {
	const sum = nums.reduce((num, acc) => num + acc, 0);
	if (sum < target || -sum > target || (sum + target) % 2 == 1) return 0;

	const n = (sum + target) / 2;
	const dp = new Array(n + 1).fill(0);
	dp[0] = 1;
	for (let num of nums) {
		for (let i = n; i >= num; i--) {
			dp[i] += dp[i - num];
		}
	}
	return dp[n];
};
