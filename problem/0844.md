/**
 * @param {string} s
 * @param {string} t
 * @return {boolean}
 */
var backspaceCompare = function (s, t) {
	return backspace(s) === backspace(t);
};

var backspace = function (str) {
	let res = '';
	for (item of str) {
		if (item === '#') res = res.slice(0, -1);
		else res += item;
	}
	return res;
};
