/**
 * @param {number[]} cards

var minimumCardPickup = function (cards) {
	let i = 0,
		n = cards.length,
		res = n + 1,
		window = new Map();
	while (i < n) {
		let card = cards[i];
		if (window.has(card)) {
			res = Math.min(res, i - window.get(card) + 1);
		}
		window.set(card, i);
		i++;
	}
	return res == n + 1 ? -1 : res;
};
