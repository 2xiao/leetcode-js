```javascript
/**
 * @param {ListNode} head
 * @return {ListNode}
 */
var oddEvenList = function (head) {
	if (!head) return head;
	let res = new ListNode(0, head);
	let odd = head;
	let even = odd.next;
	while (even && even.next) {
		let even_next = even.next;
		let odd_next = odd.next;
		even.next = even_next.next;
		even_next.next = odd_next;
		odd.next = even_next;
		odd = odd.next;
		even = even.next;
	}
	return res.next;
};
```
