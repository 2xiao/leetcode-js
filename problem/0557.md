```javascript
/**

 * @return {string}

var reverseWords = function (s) {
	return s
		.split(' ')
		.map((i) => reverse(i))
		.join(' ');
};

var reverse = function (s) {
	let arr = s.split(''),
		n = s.length;
	for (let i = 0; i < n / 2; i++) {
		[arr[i], arr[n - 1 - i]] = [arr[n - 1 - i], arr[i]];
	}
	return arr.join('');
};
```
