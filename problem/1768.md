var mergeAlternately = function (word1, word2) {
	const m = word1.length,
		n = word2.length;
	let i = 0,
		res = '';
	while (i < m && i < n) {
		res += word1[i] + word2[i];
		i++;
	}
	res += m > n ? word1.slice(i, m) : word2.slice(i, n);
	return res;
};
