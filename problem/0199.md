/**
 * @param {TreeNode} root
 * @return {number[]}

var rightSideView = function (root) {
	if (!root) return [];
	let res = [],
		queue = [root];
	while (queue.length) {
		let len = queue.length;
		res.push(queue[len - 1].val);
		for (let i = 0; i < len; i++) {
			if (queue[i].left) queue.push(queue[i].left);
			if (queue[i].right) queue.push(queue[i].right);
		}
		queue = queue.slice(len);
	}
	return res;
};
