```javascript
/**
 * @param {number[]} arr
 * @return {number[]}
 */
var arrayRankTransform = function (arr) {
	const sorted = [...new Set(arr)].sort((a, b) => a - b);

	let map = new Map();
	for (let i = 0; i < sorted.length; i++) {
		map.set(sorted[i], i + 1);
	}

	return arr.map((i) => map.get(i));
};
```
