var areAlmostEqual = function (s1, s2) {
	if (s1 == s2) return true;
	let res = [];
	for (let i = 0; i < s2.length; i++) {
		if (s1[i] !== s2[i] && res.length < 3) {
			res.push(s1[i]);
			res.push(s2[i]);
		}
	}
	return res.length == 4 && res[0] == res[3] && res[1] == res[2];
};
