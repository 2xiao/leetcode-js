var repeatedSubstringPattern = function (s) {
	const n = s.length;
	for (let i = 1; i <= n / 2; i++) {
		if (n % i == 0) {
			const str = s.slice(0, i);
			if (str.repeat(n / i) == s) {
				return true;
			}
		}
	}
	return false;
};
