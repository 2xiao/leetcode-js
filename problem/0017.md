```javascript
/**
 * @param {string} digits
 * @return {string[]}
 */
var letterCombinations = function (digits) {
	if (!digits) return [];
	const digitToLetters = {
		2: 'abc',
		3: 'def',
		4: 'ghi',
		5: 'jkl',
		6: 'mno',
		7: 'pqrs',
		8: 'tuv',
		9: 'wxyz'
	};
	let res = [];
	let track = [];

	const backtrack = (start) => {
		if (track.length == digits.length) {
			res.push(track.join(''));
			return;
		}

		for (let i of digitToLetters[digits[start]]) {
			track.push(i);
			backtrack(start + 1);
			track.pop();
		}
	};
	backtrack(0);
	return res;
};
```
