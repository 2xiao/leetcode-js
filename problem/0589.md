 * @param {Node|null} root

var preorder = function (root) {
	if (!root) return [];
	let stack = [root];
	let res = [];
	while (stack.length) {
		let node = stack.pop();
		res.push(node.val);
		for (let i = node.children.length - 1; i >= 0; i--) {
			stack.push(node.children[i]);
		}
	}
	return res;
};
