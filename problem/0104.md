var maxDepth = function (root) {

var maxDepth = function (root) {
	let track = 0;
	let res = 0;
	const backtrack = (root) => {
		if (root == null) return;

		track++;
		res = Math.max(res, track);
		backtrack(root.left);
		backtrack(root.right);
		track--;
	};
	backtrack(root);
	return res;
};
