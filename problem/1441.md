/**
 * @param {number[]} target
 * @param {number} n
 * @return {string[]}
 */
var buildArray = function (target, n) {
	let idx = 0;
	let res = [];
	for (let i = 1; i <= n; i++) {
		if (target[idx] != i) {
			res.push('Push');
			res.push('Pop');
		} else if (target[idx] === i) {
			res.push('Push');
			idx++;
		}
		if (idx === target.length) break;
	}
	return res;
};
