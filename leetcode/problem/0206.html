# 206. Reverse Linked List (opens new window)

# 题目

Reverse a singly linked list.

# 题目大意

翻转单链表

# 解题思路

有两种解题方法，一是循环、二是递归。

# 代码

# 循环

/**
 * Definition for singly-linked list.
 * function ListNode(val, next) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.next = (next===undefined ? null : next)
 * }
 */
/**
 * @param {ListNode} head
 * @return {ListNode}
 */
var reverseList = function(head) {
    let prev = null;
    let cur = head;

    while(cur !== null) {
        let next = cur.next;
        cur.next = prev;
        prev = cur;
        cur = next;
    }
    return prev;
}

# 递归


/**
 * Definition for singly-linked list.
 * function ListNode(val, next) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.next = (next===undefined ? null : next)
 * }
 */
/**
 * @param {ListNode} head
 * @return {ListNode}
 */
var reverseList = function(head) {
    if ( head === null || head.next === null) {
        return head;
    }
    const last = reverseList(head.next)
    head.next.next = head;
    head.next = null;
    return last;
};
